/* test.qu */

.language quenya 1 1

.alphabet # consonants {
	l  = "lambe"
	s  = "silme"
	t  = "tinco"
	p  = "parma"
	m  = "malta"
}

.alphabet * vowel {
	a = "tehta a"
	o = "tehta o"
	u = "tehta u"
	e = "tehta e"
	i = "tehta i"
}

.attribute part_of_speech {
	noun. = "the noun"
	verb. = "the verb"
}

.class info {
	part_of_speech
}

.match noun {
	+*
}

.match verb {
	+#
}

.match never {
	- | verb.
}

.vocabular stem

.match main {
	noun  stem | noun.
	verb  stem | verb.
	never stem | noun.
}

.target main

.vocabulary voc {
	ampa  noun.
	lasta verb.
}
